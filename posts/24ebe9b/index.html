  while(x!=link[x]){
    x=link[x];
  }
return x;
}
bool isSame(int a, int b){
  return find(a)==find(b);
}
void unite(int a, int b){
  int x = find(a);
  int y = find(b);
  //取較大的紀錄更新size,較小的更新link , 確保y較大
  if(size[x]>size[y]) swap(size[x],size[y]);
  link[x] = y; 
  size[y]+=size[x];

}

Floyd-Washall

All-pair , The algorithm maintains a two-dimensional array that contains distances between the nodes. First, distances are calculated only using direct edges between the nodes, and after this, the algorithm reduces distances by using intermediate nodes in paths. 想法為從DP來的, 考慮$D_{i,j,k}$ 表示從起點i,到終點j以1…k集合中的節點作為中間點 所能取得的最短路徑. 因此有兩種可能.

  1. 最短路徑經過k,$D_{i,j,k}=D_{i,k,k-1}+D_{k,j,k-1}$ i先走到k(用1…k-1集合中的點當作中間點)在從k到j(依樣用1…k-1集合座中間點)
  2. 最短路徑不經過k,$D_{i,j,k}=D_{i,j,k-1}$ $D_{i,j,k}=min(D_{i,j,k-1},D_{i,k,k-1}+D_{k,j,k-1})$ https://ithelp.ithome.com.tw/articles/10271531 https://www.cs.umd.edu/class/fall2017/cmsc451-0101/Lects/lect13-dp-floyd.pdf https://leetcode.com/problems/find-the-city-with-the-smallest-number-of-neighbors-at-a-threshold-distance/description/

Z-algorithm

建立一個z-array , z[i]表示字串s中的位置i起始子串s[i…n]與s 的Longest common prefix(LCP)的長度.

暴力做法為歷遍所有位置, 然後比對prefix長度 prefix i=0的時候沒有意義所以從i=1開始

vector<int> z(string s){
  int n = s.size();
  vector<int> z(n);
  
  for(int i=1;i<n;i++){
    //LCS所能用的長度必定小於n , 且目前位置的字元與從s[ z[i] ] 開始的字元相同. 
    while(i+z[i]<n && s[z[i]]==s[i+z[i]]){
      z[i]++;
    }
  return z;
  }
}

改進版本中利用 x, y 表示當前LCP已知的範圍. x為起始位置, y為終止位置. 如以計算過且prefix 的長度落在 len(x…y)以內, 此時從已知的z[i-x]開始往後計算即可

由上圖例子中 當計算超過y之後就沒有辦法利用先前計算過的z[i-x]了 

vector<int> z(string s) {

/*
定義為[start, end]區間內地k小的數字
     分區後 遞迴看pivot是地q-th 的數字
     如果q 比k大  找左半邊 k-th
     q<k , 則要找的數字在 pivot右方, q-k th
    
*/
int kselect(vector<int>&nums, int s, int e, int k){
    int pivot = nums[e];
    int small_idx =s;
    for(int i=s;i<e;i++){
        if(nums[i]<pivot){
            swap(nums[small_idx],nums[i]);
            small_idx++;
        }
    }
    swap(nums[small_idx],nums[e]);
    int q = small_idx - s + 1;  // pivot 是第 q 小
    if (k == q) {
        return nums[small_idx];
    } else if (k < q) {
        return kselect(nums, s, small_idx - 1, k); // 往左半邊找第 k 小
    } else {
        return kselect(nums, small_idx + 1, e, k - q); // 往右半邊找第 (k - q) 小
    }
}
class Solution {
public:
    int strHash(string s, int radix, int mod){
        int ret = 0;
        long factor = 1;
        for(int i=s.size()-1;i>=0;i--){
            ret = (ret+(s[i]-'a')*factor) % mod;
            factor = (factor *radix) %mod;
        }
    return ret;
    }
    int strStr(string haystack, string needle) {
        #define MOD 1000000007
        #define RADIX 26
        long long max_weight =1;
        int m = needle.size();
        int n = haystack.size();
        if(m>n) return -1;
        for(int i=0;i<m;i++) max_weight = (max_weight * RADIX) %MOD;
        long long needle_h = strHash(needle,RADIX, MOD);
        long long hay_h = 0;
        for(int i=0;i<=haystack.size()-needle.size();i++){
            if(i==0){
                hay_h = strHash(haystack.substr(0,m),RADIX,MOD);
            }else{
                hay_h = ((hay_h * RADIX) % MOD - ((haystack[i - 1] - 'a') * max_weight) % MOD + MOD) % MOD;
                hay_h = (hay_h + (haystack[i + m - 1] - 'a')) % MOD;
            }
            if(hay_h == needle_h){
                //hash code is the same check if all chars the same
                for(int j=0;j<m;j++){
                    if(needle[j]!=haystack[i+j]){
                        break;
                    }
                    if(j==m-1){
                        return i;
                    }
                }
            }
        }
    return -1;
    }
};
